def rk4(a, b, N, alpha1, alpha2):
    h = (b-a)/N
    t = a
    w1 = alpha1
    w2 = alpha2
    ret = np.array([[t, w1, w2]])
    for i in range(1,N+1):
        K11 = h * f1(t,w1, w2)
        K12 = h * f2(t,w1, w2)
        K21 = h * f1(t + h/2, w1 + K11/2, w2 + K12/2)
        K22 = h * f2(t + h/2, w1 + K11/2, w2 + K12/2)
        K31 = h * f1(t + h/2, w1 + K21/2, w2 + K22/2)
        K32 = h * f2(t + h/2, w1 + K21/2, w2 + K22/2)
        K41 = h * f1(t + h, w1 + K31, w2 + K32)
        K42 = h * f2(t + h, w1 + K31, w2 + K32)
        w1 = w1 + (K11 + 2*K21 + 2*K31 + K41)/6
        w2 = w2 + (K12 + 2*K22 + 2*K32 + K42)/6
        t = a + i*h
        ret = np.vstack((ret, np.array([t, w1, w2])))
    return ret

def pred_corr(y0=y0, a=a, b=b, h=h):
    N = int((b-a)/h)
    t = np.zeros((N+1))
    t[0] = a
    w = np.zeros((N+1))
    w[0] = y0
    
    for i in range(3):
        K1 = h * f(w[i])
        K2 = h * f(w[i] + K1/2)
        K3 = h * f(w[i] + K2/2)
        K4 = h * f(w[i] + K3)
        
        w[i+1] = w[i] + (K1 + 2*K2 + 2*K3 + K4)/6
        t[i+1] = a + (i+1)*h
        
    for i in range(4, N+1):
        t[i] = a + i*h
        w[i] = w[3] + h/2 * (3*f(w[3]) - f(w[2]))
        w[i] = w[3] + (h/12) * (5*f(w[i]) + 8*f(w[3]) - f(w[2]))
        
        for j in range(3):
            t[j] = t[j+1]
            w[j] = w[j+1]
            
        t[3] = t[i]
        w[3] = w[i]
    
    w = w[4:]
    t = t[4:]
    
    return (t,w)
